26subroutine wghpot_f(params, constants, workspace, gradphi, f)
27 !! Inputs
28 type(ddx_params_type), intent(in) :: params
29 type(ddx_constants_type), intent(in) :: constants
30 real(dp), intent(in) :: gradphi(3, constants % ncav)
31 !! Temporary buffers
32 type(ddx_workspace_type), intent(inout) :: workspace
33 !! Outputs
34 real(dp), intent(out) :: f(params % ngrid, params % nsph)
35 !! Local variables
36 integer :: isph, ig, ic, ind, ind0, jsph
37 real(dp) :: nderphi, sumSijn, sumSijn_pre
38 real(dp), dimension(3) :: vij, vtij
39 real(dp), allocatable :: SK_rijn(:), DK_rijn(:), c0(:,:), c1(:,:)
40 integer :: l0, icav, istatus, indl, inode
41 complex(dp) :: work_complex(constants % lmax0 + 1)
42 real(dp) :: work(constants % lmax0 + 1)
43
44
45 allocate(sk_rijn(0:constants % lmax0), dk_rijn(0:constants % lmax0), &
46 & c0(constants % nbasis0, params % nsph), &
47 & c1(constants % nbasis0, params % nsph), stat=istatus)
48 if (istatus.ne.0) then
49 workspace % error_flag = 1
50 workspace % error_message = 'Allocation in wghpot_f failed'
51 return
52 end if
53
54 ic = 0
55 f = zero
56 c0 = zero
57 c1 = zero
58
59 ! Compute c0 Eq.(98) QSM19.SISC
60 do isph = 1, params % nsph
61 do ig = 1, params % ngrid
62 if (constants % ui(ig, isph) .ne. zero) then
63 ic = ic + 1
64 nderphi = dot_product(gradphi(:, ic), constants % cgrid(:,ig))
65 c0(:, isph) = c0(:, isph) + &
66 & constants % wgrid(ig)*constants % ui(ig,isph)*&
67 & nderphi*constants % vgrid(1:constants % nbasis0,ig)
68 do l0 = 0, constants % lmax0
69 ind0 = l0*l0 + 1
70 ind = ind0 + 2*l0
71 c1(ind0:ind, isph) = constants % C_ik(l0, isph) * &
72 & c0(ind0:ind, isph)
73 end do
74 end if
75 end do
76 end do
77
78 ! Computation of F0 using above terms
79 ! icav: External grid poitns
80 if (params % fmm .eq. 0) then
81 icav = 0
82 do isph = 1, params % nsph
83 do ig = 1, params % ngrid
84 if (constants % ui(ig,isph).gt.zero) then
85 icav = icav + 1
86 sumsijn = zero
87 ! Loop to compute Sijn
88 do jsph = 1, params % nsph
89 sumsijn_pre = sumsijn
90 vij = params % csph(:,isph) + &
91 & params % rsph(isph)*constants % cgrid(:,ig) - &
92 & params % csph(:,jsph)
93 vtij = vij*params % kappa
94 call fmm_m2p_bessel_work(vtij, &
95 & constants % lmax0, constants % vscales, &
96 & constants % SK_ri(:, jsph), one, &
97 & c1(:, jsph), one, sumsijn, work_complex, work)
98 end do
99 ! Here Intermediate value of F_0 is computed Eq. (99)
100 ! Mutilplication with Y_lm and weights will happen afterwards
101 !write(6,*) sumSijn, epsp, eps, ddx_data % constants % ui(ig,isph)
102 f(ig,isph) = -(params % epsp/params % eps)*constants % ui(ig,isph) * sumsijn
103 end if
104 end do
105 end do
106 else
107 ! Load input harmonics into tree data
108 workspace % tmp_node_m = zero
109 workspace % tmp_node_l = zero
110 workspace % tmp_sph = zero
111 workspace % tmp_sph(1:constants % nbasis0, :) = c1(:, :)
112 if(constants % lmax0 .lt. params % pm) then
113 do isph = 1, params % nsph
114 inode = constants % snode(isph)
115 workspace % tmp_node_m(1:constants % nbasis0, inode) = &
116 & workspace % tmp_sph(1:constants % nbasis0, isph)
117 workspace % tmp_node_m(constants % nbasis0+1:, inode) = zero
118 end do
119 else
120 indl = (params % pm+1)**2
121 do isph = 1, params % nsph
122 inode = constants % snode(isph)
123 workspace % tmp_node_m(:, inode) = workspace % tmp_sph(1:indl, isph)
124 end do
125 end if
126 ! Do FMM operations
127 call tree_m2m_bessel_rotation(params, constants, workspace % tmp_node_m)
128 call tree_m2l_bessel_rotation(params, constants, workspace % tmp_node_m, &
129 & workspace % tmp_node_l)
130 call tree_l2l_bessel_rotation(params, constants, workspace % tmp_node_l)
131 call tree_l2p_bessel(params, constants, one, workspace % tmp_node_l, zero, &
132 & workspace % tmp_grid)
133 call tree_m2p_bessel(params, constants, constants % lmax0, one, &
134 & params % lmax, workspace % tmp_sph, one, &
135 & workspace % tmp_grid)
136 f = -(params % epsp/params % eps) * constants % ui * workspace % tmp_grid
137 end if
138
139 deallocate(sk_rijn, dk_rijn, c0, c1)
140
141end subroutine wghpot_f

151subroutine calcv2_lpb (params, constants, isph, pot, x)
152 type(ddx_params_type), intent(in) :: params
153 type(ddx_constants_type), intent(in) :: constants
154 integer, intent(in) :: isph
155 real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: x
156 real(dp), dimension(params % ngrid), intent(inout) :: pot
157 complex(dp) :: work_complex(params % lmax+1)
158 real(dp) :: work(params % lmax+1)
159 real(dp), dimension(params % ngrid) :: pot2
160 integer :: its, ij, jsph
161 real(dp) :: vij(3), sij(3), vtij(3)
162 real(dp) :: vvij, tij, xij, oij
163
164 pot = zero
165 pot2 = zero
166 do its = 1, params % ngrid
167 if (constants % ui(its,isph).lt.one) then
168 do ij = constants % inl(isph), constants % inl(isph+1)-1
169 jsph = constants % nl(ij)
170
171 ! compute geometrical variables
172 vij = params % csph(:,isph) + params % rsph(isph)*constants % cgrid(:,its) - params % csph(:,jsph)
173 vvij = sqrt(dot_product(vij,vij))
174 tij = vvij/params % rsph(jsph)
175
176 if ( tij.lt.( one + (params % se+one)/two*params % eta ) ) then
177 sij = vij/vvij
178 xij = fsw(tij, params % se, params % eta)
179 if (constants % fi(its,isph).gt.one) then
180 oij = xij/constants % fi(its, isph)
181 else
182 oij = xij
183 end if
184 vtij = vij*params % kappa
185 call fmm_l2p_bessel_work(vtij, &
186 & params % lmax, constants % vscales, &
187 & constants % SI_ri(:, jsph), oij, x(:, jsph), one, &
188 & pot(its), work_complex, work)
189 end if
190 end do
191 end if
192 end do
193endsubroutine calcv2_lpb

202subroutine convert_ddcosmo(params, constants, direction, vector)
203 !! Inputs
204 type(ddx_params_type), intent(in) :: params
205 type(ddx_constants_type), intent(in) :: constants
206 integer, intent(in) :: direction
207 !! Outputs
208 real(dp), intent(inout) :: vector(constants % nbasis, params % nsph)
209 !! Local variables
210 integer :: isph, l, m, ind
211 real(dp) :: fac
212 !! The code
213 do isph = 1, params % nsph
214 do l = 0, params % lmax
215 ind = l*l + l + 1
216 fac = fourpi / (two*dble(l) + one)
217 if (direction .eq. -1) fac = one / fac
218 do m = -l, l
219 vector(ind+m, isph) = fac*vector(ind+m, isph)
220 end do
221 end do
222 end do
223end subroutine convert_ddcosmo

237subroutine inthsp(params, constants, rijn, isph, basloc, fac_hsp, work)

274subroutine adjrhs_lpb(params, constants, isph, xi, vlm, basloc, &
275 & vplm, vcos, vsin, tmp_bessel)
276 implicit none
277 !! input/output
278 type(ddx_params_type), intent(in) :: params
279 type(ddx_constants_type), intent(in) :: constants
280 real(dp), dimension(params % ngrid, params % nsph), intent(in) :: xi
281 real(dp), dimension((params % lmax+1)**2), intent(out) :: vlm
282 real(dp), dimension((params % lmax+1)**2), intent(out) :: basloc, vplm
283 real(dp), dimension(params % lmax+1), intent(out) :: vcos, vsin
284 complex(dp), dimension(max(2, params % lmax+1)), intent(out) :: tmp_bessel
285 integer, intent(in) :: isph
286 !! local
287 integer :: ij, jsph, ig, l, ind, m
288 real(dp) :: vji(3), vvji, tji, sji(3), xji, oji, fac
289 real(dp) :: rho, ctheta, stheta, cphi, sphi
290 real(dp), dimension(constants % nbasis) :: fac_hsp
291
292 !loop over neighbors of i-sphere
293 do ij = constants % inl(isph), constants % inl(isph+1)-1
294 !j-sphere is neighbor
295 jsph = constants % nl(ij)
296 !loop over integration points
297 do ig = 1, params % ngrid
298 !compute t_n^ji = | r_j + \rho_j s_n - r_i | / \rho_i
299 vji = params % csph(:,jsph) + params % rsph(jsph)* &
300 & constants % cgrid(:,ig) - params % csph(:,isph)
301 vvji = sqrt(dot_product(vji,vji))
302 tji = vvji/params % rsph(isph)
303 !point is INSIDE i-sphere (+ transition layer)
304 if ( tji.lt.( one + (params % se+one)/two*params % eta ) ) then
305 !compute s_n^ji
306 sji = vji/vvji
307 call ylmbas(sji, rho, ctheta, stheta, cphi, &
308 & sphi, params % lmax, &
309 & constants % vscales, basloc, &
310 & vplm, vcos, vsin)
311 call inthsp_adj(params, constants, vvji, isph, basloc, fac_hsp, &
312 & tmp_bessel)
313 !compute \chi( t_n^ji )
314 xji = fsw( tji, params % se, params % eta )
315 !compute W_n^ji
316 if ( constants % fi(ig,jsph).gt.one ) then
317 oji = xji/ constants % fi(ig,jsph)
318 else
319 oji = xji
320 endif
321 !compute w_n * xi(n,j) * W_n^ji
322 fac = constants % wgrid(ig) * xi(ig,jsph) * oji
323 !loop over l
324 do l = 0, params % lmax
325 ind = l*l + l + 1
326 !loop over m
327 do m = -l,l
328 vlm(ind+m) = vlm(ind+m) + fac*fac_hsp(ind+m)
329 enddo
330 enddo
331 endif
332 enddo
333 enddo
334end subroutine adjrhs_lpb
