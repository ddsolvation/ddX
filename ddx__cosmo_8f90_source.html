205 type(ddx_constants_type), intent(in) :: constants

207 type(ddx_state_type), intent(inout) :: state
208 type(ddx_error_type), intent(inout) :: ddx_error

248subroutine cosmo_derivative_setup(params, constants, workspace, state)
249 type(ddx_params_type), intent(in) :: params
250 type(ddx_constants_type), intent(in) :: constants
251 type(ddx_workspace_type), intent(inout) :: workspace
252 type(ddx_state_type), intent(inout) :: state
253
254 real(dp), external :: ddot
255 integer :: icav, isph, igrid
256
257 ! dummy operation on unused interface arguments
258 if (allocated(workspace % tmp_pot)) continue
259
260 ! Get values of S on the grid
261 call ddeval_grid_work(constants % nbasis, params % ngrid, params % nsph, &
262 & constants % vgrid, constants % vgrid_nbasis, one, state % s, zero, &
263 & state % sgrid)
264 ! Get the values of phi on the grid
265 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
266 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
267 & state % phi_grid)
268
269 ! assemble the intermediate zeta: S weighted by U evaluated on the
270 ! exposed grid points.
271 icav = 0
272 do isph = 1, params % nsph
273 do igrid = 1, params % ngrid
274 if (constants % ui(igrid, isph) .ne. zero) then
275 icav = icav + 1
276 state % zeta(icav) = constants % wgrid(igrid) * &
277 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
278 & constants % vgrid(1, igrid), 1, state % s(1, isph), 1)
279 end if
280 end do
281 end do
282
283end subroutine cosmo_derivative_setup
