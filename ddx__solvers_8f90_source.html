80subroutine jacobi_diis(params, constants, workspace, tol, rhs, x, niter, &
81 & x_rel_diff, matvec, dm1vec, norm_func, info)
82 !! Inputs
83 type(ddx_params_type), intent(in) :: params
84 type(ddx_constants_type), intent(in) :: constants
85 real(dp), intent(in) :: tol, rhs(constants % n)
86 !! Temporaries
87 type(ddx_workspace_type), intent(inout) :: workspace
88 !! Outputs
89 real(dp), intent(inout) :: x(constants % n)
90 integer, intent(inout) :: niter
91 real(dp), intent(out) :: x_rel_diff(niter)
92 integer, intent(out) :: info
93 !! External procedures
94 procedure(matvec_interface) :: matvec, dm1vec
95 procedure(norm_interface) :: norm_func
96 !! Local variables
97 integer :: it, nmat
98 real(dp) :: diff, norm, rel_diff
99 !! The code
100 ! DIIS variable
101 nmat = 1
102 ! Iterations
103 do it = 1, niter
104 ! y = rhs - O x
105 call matvec(params, constants, workspace, x, workspace % tmp_y)
106 workspace % tmp_y = rhs - workspace % tmp_y
107 ! x_new = D^-1 y
108 call dm1vec(params, constants, workspace, workspace % tmp_y, &
109 & workspace % tmp_x_new)
110 ! DIIS extrapolation
111 if (params % jacobi_ndiis .gt. 0) then
112 workspace % tmp_x_diis(:, nmat) = workspace % tmp_x_new
113 workspace % tmp_e_diis(:, nmat) = workspace % tmp_x_new - x
114 call diis(constants % n, nmat, params % jacobi_ndiis, &
115 & workspace % tmp_x_diis, workspace % tmp_e_diis, &
116 & workspace % tmp_bmat, workspace % tmp_x_new)
117 end if
118 ! Norm of increment
119 x = workspace % tmp_x_new - x
120 diff = norm_func(params % lmax, constants % nbasis, params % nsph, x)
121 norm = norm_func(params % lmax, constants % nbasis, params % nsph, &
122 & workspace % tmp_x_new)
123
124 ! compute rel_diff using the rule 0/0 = 0
125 if (diff.eq.zero .and. norm.eq.zero) then
126 rel_diff = zero
127 else
128 rel_diff = diff / norm
129 end if
130
131 x_rel_diff(it) = rel_diff
132 ! Update solution
133 x = workspace % tmp_x_new
134 ! Check stop condition
135 if (rel_diff .le. tol) then
136 info = 0
137 niter = it
138 return
139 end if
140 end do
141 info = 1
142endsubroutine jacobi_diis

243!*********************************************************
244! GMRESR algorithm to solve linear system Ax = b
245!
246! author:
247! m.botchev, utrecht university, december 1996
248! report bugs to botchev@cwi.nl or botchev@excite.com
249
250! Copyright (c) 1996 by M.A.Botchev
251! Permission to copy all or part of this work is granted,
252! provided that the copies are not made or distributed
253! for resale, and that the copyright notice and this
254! notice are retained.
255
256! Details to the algorithm may be found in
257! H.A. van der Vorst, !. Vuik, "GMRESR: a Family of Nested GMRES
258! Methods", Num. Lin. Alg. Appl., vol. 1(4), 369--386 (1994)
259
260! parameter list:
261! oktest is TRUE if intermediate residuals should be printed
262! n == INTEGER size of problem
263! j == INTEGER truncation parameter (work with j last vectors)
264! mgmres == INTEGER dimension of the envoked GMRES
265! if mgmres.eq.0 then we get GCR algorithm, the simplest
266! version of GMRESR
267! b == DOUBLE PRECISION righthand side vector
268! x == DOUBLE PRECISION initial guess on input,
269! (approximate) solution on output
270! work == DOUBLE PRECISION work space 1 of size n x (2*j+mgmres+2)
271! eps == DOUBLE PRECISION tolerance of stopping criterion.
272! process is stopped as soon as
273! (1) residual norm has been dumped by factor eps,
274! i.e. ||res|| / ||b|| <= eps OR
275! (2) maximum number of iterations maxit has been performed
276! stc == CHARACTER*3
277! Determine stopping criterion (||.|| denotes the 2-norm):
278! stc='rel' -- relative stopping crit.: ||res|| < eps*||res0||
279! stc='abs' -- absolute stopping crit.: ||res|| < eps
280! maxits == INTEGER max. no. outer_iterative_steps/truncation_length on input
281! on output it is the actual number of total iterative steps
282! resid == DOUBLE PRECISION residual measure (depends on stopping criterion)
283! achieved on output
284! iflag == INTEGER on output 0 - solution found within tolerance
285! 1 - no convergence within niter
286
287subroutine gmresr(params, constants, workspace, eps, b, x, &
288 & niter, resid, matvec, info)
289 !! Inputs
290 type(ddx_params_type), intent(in) :: params
291 type(ddx_constants_type), intent(in) :: constants
292 real(dp), intent(in) :: eps, b(constants % n)
293 !! Outputs
294 real(dp), intent(inout) :: x(constants % n)
295 integer, intent(inout) :: niter
296 !real(dp), intent(out) :: x_rel_diff(niter)
297 integer, intent(out) :: info
298 !! Temporary buffers
299 type(ddx_workspace_type), intent(inout) :: workspace
300 !! External procedures
301 procedure(matvec_interface) :: matvec
302 !! Local variables
303
304! list of variables: arrays in alphabetical order,
305! then other variables in alphabetical order
306
307 integer i,its,nits,itsinn,k,n,j
308
309 double precision alpha, alphai, cknorm, ckres, ddot, dnrm2, &
310 & epsinn, res0,resnor,resid
311
312! distribute work space work(n,*) among some virtual variables;
313! namely, we think of columns of work as being occupied by
314! c(n,0:j-1), u(n,0:j-1), resid(n), workgmr(n,params % gmresr_dim+1)
315! therefore we define "shifts"
316
317 integer c, u, workres, workgmre
318! ----------------------------------------------------------------------------
319
320 ! Read j and n from the parameters
321 j = params % gmresr_j
322 n = constants % n
323! c occupies j columns 0...j-1:
324 c = 0
325! u occupies j columns j...2*j-1:
326 u = params % gmresr_j
327! resid occupies 1 column No. 2*j:
328 workres = 2*j
329! workgmre occupies gmres_dim+1 columns 2*j+1...2*j+gmres_dim+1:
330 workgmre = 2*j+1
331! so that we can access, say, to the (k)th column of the virtual
332! array c(n,0:j-1) as work(1,c+k),
333! virtual residual vector resid(n) is work(1,workres) and so on ...
334
335! ***Furthermore, we build sequences c_k and u_k, k = 0,...,m-1
336! but we store only the last j vectors in the following way:
337! Assume j=3, then
338! --------------------------------------------------------------
339! k | number of column of work | columns of work which are vectors
340! | in which we store c_k | we actually store and use
341! 0 | 0 | c_0 u_0 ...
342! 1 | 1 | c_0 c_1 u_0 u_1 ...
343! 2 | 2 | c_0 c_1 c_2 u_0 u_1 u_2 ...
344! 3 | 0 | c_3 c_1 c_2 u_3 u_1 u_2 ...
345! 4 | 1 | c_3 c_4 c_2 u_3 u_4 u_2 ...
346! 5 | 2 | c_3 c_4 c_5 u_3 u_4 u_5 ...
347! 6 | 0 | c_6 c_4 c_5 u_6 u_4 u_5 ...
348! ... | ... | ... ...
349! This mapping is performed by function mod(k,j)
350!
351
352! Reset iteration counter
353 nits= 0
354 its = 0
355! Calculate (initial) residual norm
356 call matvec(params, constants, workspace, x, &
357 & workspace % tmp_gmresr(1, workres))
358 alpha = -1
359!
360 call daxpy(n, alpha, b, 1, workspace % tmp_gmresr(1, workres), 1)
361 call dscal(n, alpha, workspace % tmp_gmresr(1, workres), 1)
362
363! Calculate residual norm and quit if it is zero
364 res0 = dnrm2(n, b, 1)
365 resnor = res0
366 resid = 0
367
368 if ( res0 .eq. 0.0d0 ) then
369 info = 0
370 niter = 0
371 return
372 end if
373
374 resid=resnor/res0
375
376 if ( resid .le. eps ) then
377 info = 0
378 niter = 0
379 return
380 end if
381
382! Main iterative loop ============================
383 k = -1
384 do while (.true.)
385
386! Loop to increment dimension of projection subspace
387 k=k+1
388
389! Number of step (not restart) to be done
390 its = its + 1
391! write(*,'(A,i3)') '+++++++++++++++++ its ',its
392
393! - - - - - - - - - - - - - - - - - - - - - - - - -
394! This part should deliver
395! u(1,k) <-- invA * resid
396! where u(1,k) is the k-th column of array u(1:n,0:m) and
397! invA is some reasonable approximation to the inverse of A
398!
399! If gmres_dim=0 then no inner iterations are performed
400! to get invA, so that invA is just the identity matrix.
401! In this case algorithm GMRESR is nothing but GCR
402!
403! Otherwise for inner iterations we perform ONE restart of GMRES
404! ATTN: for this implementation it is crucial to perform only
405! ONE restart of GMRES
406 if (params % gmresr_dim .eq. 0) then
407! u(1,k) := resid
408 call dcopy(n, workspace % tmp_gmresr(1, workres), 1, &
409 & workspace % tmp_gmresr(1, u+mod(k,j)), 1)
410 call matvec(params, constants, workspace, &
411 & workspace % tmp_gmresr(1, u+mod(k,j)), &
412 & workspace % tmp_gmresr(1, c+mod(k,j)))
413 nits=nits+1
414 else
415! Solve linear system A*u(1,k)=resid by GMRES
416! The stopping criterion for inner iterations is
417! always absolute but it is automatically adjusted
418! not to be stricter than the stopping criterion for the
419! outer iterations. For example, if stop.criterion for
420! the outer iterations is relative than absolute stop.
421! criterion for the inner iterations is (eps*res0)
422! Accuracy for inner iteration:
423
424 epsinn = eps*res0
425
426! After envoking gmres0 epsinn and itsinn contain actual achieved
427! accuracy and number of performed iterations respectively
428
429 itsinn=params % gmresr_dim
430
431 call gmres0(params, constants, workspace, n, &
432 & workspace % tmp_gmresr(1, workres), &
433 & workspace % tmp_gmresr(1, u+mod(k,j)), &
434 & workspace % tmp_gmresr(1, c+mod(k,j)), &
435 & workspace % tmp_gmresr(1, workgmre), &
436 & epsinn,itsinn,matvec)
437 !write(*,*) 'GMRES ', work(1,c+mod(k,j))
438
439 nits=nits+itsinn
440 end if
441! - - - - - - - - - - - - - - - - - - - - - - - -
442
443! Inner loop to orthogonalize
444! c(1,k) with respect to c(1,k-j),...,c(1,k-1)
445! and to update correspondingly
446! u(1,k) with respect to u(1,k-j),...,u(1,k-1)
447! parameter j is used only here
448 do i = max0(0,k-j),k-1
449 alphai = ddot(n, workspace % tmp_gmresr(1, c+mod(i,j)), 1, &
450 & workspace % tmp_gmresr(1, c+mod(k,j)), 1)
451 call daxpy(n, -alphai, workspace % tmp_gmresr(1, c+mod(i,j)), 1, &
452 & workspace % tmp_gmresr(1, c+mod(k,j)), 1)
453 call daxpy(n, -alphai, workspace % tmp_gmresr(1, u+mod(i,j)), 1, &
454 & workspace % tmp_gmresr(1, u+mod(k,j)), 1)
455 end do
456
457! Normalize c(1,k) and "normalize" u(1,k)
458 cknorm = dnrm2(n, workspace % tmp_gmresr(1, c+mod(k,j)), 1)
459 cknorm = 1 / cknorm
460 call dscal(n, cknorm, workspace % tmp_gmresr(1, c+mod(k,j)), 1)
461 call dscal(n, cknorm, workspace % tmp_gmresr(1, u+mod(k,j)), 1)
462
463! Update current solution and residual
464 ckres = ddot(n, workspace % tmp_gmresr(1, c+mod(k,j)), 1, &
465 & workspace % tmp_gmresr(1, workres), 1)
466 call daxpy(n, ckres, workspace % tmp_gmresr(1, u+mod(k,j)), 1, x, 1)
467 call daxpy(n, -ckres, workspace % tmp_gmresr(1, c+mod(k,j)), 1, &
468 & workspace % tmp_gmresr(1, workres), 1)
469
470
471! call show(n,10,x,'GMRESR ')
472
473! Calculate residual norm, check convergence
474 resnor = dnrm2(n, workspace % tmp_gmresr(1, workres), 1)
475
476 resid=resnor/res0
477
478 if ( resid .le. eps ) then
479 info = 0
480 niter = nits
481 return
482 end if
483 if (its .ge. niter*j) then
484 info = 1
485 niter = nits
486 return
487 end if
488
489! print 11, ' ||res|| = ',resnor
490! 11 format(A,d)
491! 13 format(i4,A,d)
492
493 end do
494! End of inifinite iterative loop =================
495! End of GMRESR subroutine
496 end
497! This is the modified GMRES routine gmres0 adapted for GMRESR by
498! Mike Botchev, Utrecht University, Dec. 1996
499! For detail on how to make GMRES (for GMRESR) cheaper see
500! the above-mentioned paper on GMRESR
501!*************************************************************
502! This code was initially written by Youcef Saad
503! then revised by Henk A. van der Vorst
504! and Mike Botchev (oct. 1996)
505! ************************************************************
506! gmres algorithm . simple version . (may 23, 1985)
507! parameter list:
508! oktest == TRUE for printing intermediate results
509! n == size of problem
510! im == size of krylov subspace: should not exceed 50 in this
511! version (can be reset in code. looking at comment below)
512! rhs == right hand side
513! uu == initial guess for vector u (see above-mentioned paper on GMRESR)
514! on input, approximate solution on output
515! cc == initial guess for vector c (see above-mentioned paper on GMRESR)
516! on input, approximate solution on output
517! work0 == work space of size n x (im+1)
518! eps == tolerance for stopping criterion. process is stopped
519! as soon as ( ||.|| is the euclidean norm):
520! || current residual || <= eps
521! maxits == maximum number of iterations allowed
522! on OUTPUT: actual number of iterations performed
523! ----------------------------------------------------------------
524! subroutines
525! matvec == matrix vector multiplication y <- A*x
526
527! BLAS:
528! dcopy == y <-- x routine
529! ddot == dot product function
530! dnrm2 == euclidean norm function
531! daxpy == y <-- y+ax routine
532! dscal == x <-- ax routine
533! dtsrv == to solve linear system with a triangular matrix
534!*************************************************************
535!-------------------------------------------------------------
536! arnoldi size should not exceed 10 in this version..
537! to reset modify maxdim. BUT: ----------------
538! maxdim was set to 10 because of two reasons:
539! (1) it is assumed in this implementation that it is cheaper to
540! make maxdim vector updates than to make 1 matrix-vector
541! multiplication;
542! (2) for large maxdim we may lose the orthogonality property
543! on which this cheap implementation is based.
544! Please keep it in mind changing maxdim
545!-------------------------------------------------------------
546!=============================================================================
547subroutine gmres0(params, constants, workspace, n, rhs, uu, cc, work0, eps, niter, matvec)
548
549 !! Inputs
550 type(ddx_params_type), intent(in) :: params
551 type(ddx_constants_type), intent(in) :: constants
552 !! Temporary buffers
553 type(ddx_workspace_type), intent(inout) :: workspace
554 integer maxdim,maxd1,md1max
555 parameter(maxdim=20, maxd1=maxdim+1, md1max=maxdim*maxd1)
556 procedure(matvec_interface) :: matvec
557
558 integer jjj,jj1
559 integer i,i1,its,j,k,k1,niter,n
560 double precision cc(n),coeff,coef1,dabs,ddot,dnrm2,dsqrt,eps,epsmac, &
561 & gam,rhs(n),ro,uu(n),work0(n,params % gmresr_dim+1),t
562
563 double precision hh(maxd1,maxdim),hh1(maxd1,maxdim),c(maxdim), &
564 & s(maxdim),rs(maxd1),rs1(maxd1)
565
566 data (( hh(jj1,jjj), jj1=1,maxd1), jjj=1,maxdim) / md1max*0.0 / ,&
567 & epsmac / 1.d-16 /
568 its = 0
569
570! ----------------------------
571! Outer loop starts here..
572! BUT here (for GMRESR) only 1 outer loop is allowed
573! Compute initial residual vector
574! ----------------------------
575 10 continue
576! do not calculate initial residual first restart because
577! initial guess is always zero.
578! make initial guess zero:
579 coeff = 0.0
580 call dscal(n,coeff,uu,1)
581! make initial residual right-hand side:
582 call dcopy(n,rhs,1,work0,1)
583
584 ro = dnrm2(n, work0, 1)
585 if ((ro .eq. 0.0d0).or.(ro .le. eps)) then
586 call matvec(params, constants, workspace, uu, cc)
587 eps = ro
588 niter = its
589 return
590 end if
591
592 coeff = 1 / ro
593 call dscal(n,coeff,work0,1)
594
595! initialize 1-st term of rhs of hessenberg system..
596 rs(1) = ro
597 i = 0
598
599 4 continue
600 i=i+1
601 its = its + 1
602 i1 = i + 1
603 call matvec(params, constants, workspace, work0(1,i), work0(1,i1))
604! -----------------------------------------
605! modified gram - schmidt...
606! -----------------------------------------
607 do j=1, i
608 t = ddot(n, work0(1,j),1,work0(1,i1),1)
609 hh(j,i) = t
610 call daxpy(n, -t, work0(1,j), 1, work0(1,i1), 1)
611 end do
612 t = dnrm2(n, work0(1,i1), 1)
613 hh(i1,i) = t
614 if (t .ne. 0.0d0)then
615 t = 1 / t
616 call dscal(n, t, work0(1,i1), 1)
617! save new column of hh in hh1 to reproduce vector cc later on
618 call dcopy(maxd1,hh(1,i),1,hh1(1,i),1)
619 endif
620! done with modified gram schmidt and arnoldi step..
621
622! now update factorization of hh
623 if (i .ne. 1) then
624! perform previous transformations on i-th column of h
625 do k=2,i
626 k1 = k-1
627 t = hh(k1,i)
628 hh(k1,i) = c(k1)*t + s(k1)*hh(k,i)
629 hh(k,i) = -s(k1)*t + c(k1)*hh(k,i)
630 end do
631 endif
632 gam = dsqrt(hh(i,i)**2 + hh(i1,i)**2)
633 if (gam .eq. 0.0d0) gam = epsmac
634
635! determine next plane rotation
636 c(i) = hh(i,i)/gam
637 s(i) = hh(i1,i)/gam
638 rs(i1) = -s(i)*rs(i)
639 rs(i) = c(i)*rs(i)
640
641! determine residual norm and test for convergence-
642 hh(i,i) = c(i)*hh(i,i) + s(i)*hh(i1,i)
643 ro = dabs(rs(i1))
644 if ((i .lt. params % gmresr_dim) .and. (ro .gt. eps)) goto 4
645
646! now compute solution. first solve upper triangular system.
647
648! rs := hh(1:i,1:i) ^-1 * rs
649
650 call dtrsv('U','N','N',i,hh,maxd1,rs,1)
651! done with back substitution..
652
653! now form linear combination to get vector uu
654 do j=1, i
655 t = rs(j)
656 call daxpy(n, t, work0(1,j), 1, uu,1)
657 end do
658! DO NOT restart outer loop EVEN when necessary (that is crucial
659! for this implementation of GMRESR): NEVER goto 10 !
660! if (ro .gt. eps .and. its .lt. niter) goto 10
661
662! Finally, reproduce vector cc as cc = A*uu = work0*hh1*rs:
663! rs := hh1(1:i1,1:i) * rs
664 coeff = 1
665 coef1 = 0
666 call dgemv('N',i1,i,coeff,hh1,maxd1,rs,1,coef1,rs1,1)
667
668! now multiply Krylov basis vectors work0 by rs:
669! cc := work0*rs
670 call dscal(n,coef1,cc,1)
671 do j=1, i1
672 t = rs1(j)
673 call daxpy(n, t, work0(1,j), 1, cc,1)
674 end do
675
676 niter=its
677 eps=ro
678 return
679!------------------------------- end of gmres0 ----------------------
680 end

682! jacobi_diis_solver_external
683!
684! WARNING: code duplication is bad, but here it makes a very specific sense.
685!
686! in ddLPB, we are solving a linear system with Jacobi DIIS also to apply the
687! preconditioner. As calling a solver recursively is a terrible idea, we
688! create a copy of jacobi_diis that
689!
690! - takes as an input the size of the system to be solved (no hardcoded dependence
691! on nsph)
692! - uses copies of the norm subroutine, so that it can also be called with
693! a user-given size for arrays
694!
695subroutine jacobi_diis_external(params, constants, workspace, n, tol, rhs, x, n_iter, &
696 & x_rel_diff, matvec, dm1vec, norm_func, info)
697 type(ddx_params_type), intent(in) :: params
698 type(ddx_constants_type), intent(in) :: constants
699 type(ddx_workspace_type), intent(inout) :: workspace
700 ! Inputs
701 integer, intent(in) :: n
702 real(dp), intent(in) :: tol
703 real(dp), dimension(n), intent(in) :: rhs
704 ! Outputs
705 real(dp), dimension(n), intent(inout) :: x
706 integer, intent(inout) :: n_iter, info
707 real(dp), intent(out) :: x_rel_diff(n_iter)
708!
709 external :: matvec, dm1vec
710 procedure(norm_interface) :: norm_func
711 ! Local variables
712 integer :: it, nmat, istatus, lenb, nsph_u
713 real(dp) :: diff, norm, rel_diff = zero
714 logical :: dodiis
715!
716 real(dp), allocatable :: x_new(:), y(:), x_diis(:,:), e_diis(:,:), bmat(:,:)
717!
718!---------------------------------------------------------------------------------------------
719!
720!
721! DIIS extrapolation flag
722 dodiis = (params%jacobi_ndiis.gt.0)
723!
724! extrapolation required
725 if (dodiis) then
726!
727! allocate workspaces
728 lenb = params % jacobi_ndiis + 1
729 allocate( x_diis(n,params % jacobi_ndiis), e_diis(n,params % jacobi_ndiis), bmat(lenb,lenb) , stat=istatus )
730 if (istatus .ne. 0) then
731 write(*,*) ' jacobi_diis: [1] failed allocation (diis)'
732 stop
733 endif
734!
735! initialize the number of points for diis to one.
736! note that nmat is updated by diis.
737 nmat = 1
738!
739 endif
740!
741! allocate workspaces
742 allocate( x_new(n), y(n) , stat=istatus )
743 if (istatus .ne. 0) then
744 write(*,*) ' jacobi_diis: [2] failed allocation (scratch)'
745 stop
746 endif
747!
748! Jacobi iterations
749! =================e
750 do it = 1, n_iter
751!
752! y = rhs - O x
753 call matvec(params, constants, workspace, x, y )
754 y = rhs - y
755!
756! x_new = D^-1 y
757 call dm1vec(params, constants, workspace, y, x_new)
758!
759! DIIS extrapolation
760! ==================
761 if (dodiis) then
762!
763 x_diis(:,nmat) = x_new
764 e_diis(:,nmat) = x_new - x
765!
766 call diis(n,nmat,params % jacobi_ndiis,x_diis,e_diis,bmat,x_new)
767!
768 endif
769!
770! increment
771 x = x_new - x
772!
773! warning: in the following, we use a quick and dirty hack to pass
774! to the norm function arrays double the size than in non ddlpb calculations
775! by defining a fake number of spheres with is n/nbasis (i.e., 2*nsph in
776! ddlpb).
777!
778 nsph_u = n/constants % nbasis
779 diff = norm_func(params % lmax, constants % nbasis, nsph_u, x)
780 norm = norm_func(params % lmax, constants % nbasis, nsph_u, x_new)
781 ! compute rel_diff using the rule 0/0 = 0
782 if (diff.eq.zero .and. norm.eq.zero) then
783 rel_diff = zero
784 else
785 rel_diff = diff / norm
786 end if
787
788 x_rel_diff(it) = rel_diff
789!
790! update
791 x = x_new
792!
793! EXIT Jacobi loop here
794! =====================
795 if (rel_diff .le. tol) then
796 info = 0
797 n_iter = it
798 return
799 end if
800!
801 enddo
802!
803! something went wrong...
804 info = 1
805!
806 return
807!
808!
809endsubroutine jacobi_diis_external
