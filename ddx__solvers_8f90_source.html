79subroutine jacobi_diis(params, constants, workspace, tol, rhs, x, niter, &
80 & x_rel_diff, matvec, dm1vec, norm_func)
81 !! Inputs
82 type(ddx_params_type), intent(in) :: params
83 type(ddx_constants_type), intent(in) :: constants
84 real(dp), intent(in) :: tol, rhs(constants % n)
85 !! Temporaries
86 type(ddx_workspace_type), intent(inout) :: workspace
87 !! Outputs
88 real(dp), intent(inout) :: x(constants % n)
89 integer, intent(inout) :: niter
90 real(dp), intent(out) :: x_rel_diff(niter)
91 !! External procedures
92 procedure(matvec_interface) :: matvec, dm1vec
93 procedure(norm_interface) :: norm_func
94 !! Local variables
95 integer :: it, nmat
96 real(dp) :: diff, norm, rel_diff
97 !! The code
98 ! DIIS variable
99 nmat = 1
100 ! Iterations
101 do it = 1, niter
102 ! y = rhs - O x
103 call matvec(params, constants, workspace, x, workspace % tmp_y)
104 workspace % tmp_y = rhs - workspace % tmp_y
105 ! x_new = D^-1 y
106 call dm1vec(params, constants, workspace, workspace % tmp_y, &
107 & workspace % tmp_x_new)
108 ! DIIS extrapolation
109 if (params % jacobi_ndiis .gt. 0) then
110 workspace % tmp_x_diis(:, nmat) = workspace % tmp_x_new
111 workspace % tmp_e_diis(:, nmat) = workspace % tmp_x_new - x
112 call diis(constants % n, nmat, params % jacobi_ndiis, &
113 & workspace % tmp_x_diis, workspace % tmp_e_diis, &
114 & workspace % tmp_bmat, workspace % tmp_x_new)
115 end if
116 ! Norm of increment
117 x = workspace % tmp_x_new - x
118 diff = norm_func(params % lmax, constants % nbasis, params % nsph, x)
119 norm = norm_func(params % lmax, constants % nbasis, params % nsph, &
120 & workspace % tmp_x_new)
121
122 ! compute rel_diff using the rule 0/0 = 0
123 if (diff.eq.zero .and. norm.eq.zero) then
124 rel_diff = zero
125 else
126 rel_diff = diff / norm
127 end if
128
129 if (params % verbose) then
130 write(params % iunit, '(A,I3,X,A,E10.2)') &
131 & 'Iteration:', it, 'Rel. diff:', rel_diff
132 flush(params % iunit)
133 end if
134
135 x_rel_diff(it) = rel_diff
136 ! Update solution
137 x = workspace % tmp_x_new
138 ! Check stop condition
139 if (rel_diff .le. tol) then
140 niter = it
141 return
142 end if
143 end do
144 workspace % error_flag = 1
145 workspace % error_message = "Jacobi solver did not converge"
146 return
147endsubroutine jacobi_diis

150subroutine diis(n, nmat, ndiis, x, e, b, xnew)
151 implicit none
152 integer, intent(in) :: n, ndiis
153 integer, intent(inout) :: nmat
154 real(dp), dimension(n,ndiis), intent(inout) :: x, e
155 real(dp), dimension(ndiis+1,ndiis+1), intent(inout) :: b
156 real(dp), dimension(n), intent(inout) :: xnew
157 integer :: nmat1, i, istatus, info
158 integer :: j, k
159 real(dp), allocatable :: bloc(:,:), cex(:)
160 integer, allocatable :: ipiv(:)
161 real(dp), parameter :: zero = 0.0d0, one = 1.0d0
162 integer, parameter :: delta = 10
163
164 if (nmat.ge.ndiis) then
165 do j = 2, nmat - delta
166 do k = 2, nmat - delta
167 b(j, k) = b(j+delta, k+delta)
168 end do
169 end do
170 do j = 1, nmat - delta
171 x(:, j) = x(:, j+delta)
172 e(:, j) = e(:, j+delta)
173 end do
174 nmat = nmat - delta
175 end if
176 nmat1 = nmat + 1
177
178 allocate(bloc(nmat1, nmat1), cex(nmat1), ipiv(nmat1), stat=istatus)
179
180 call makeb(n, nmat, ndiis, e, b)
181 bloc = b(1:nmat1, 1:nmat1)
182 cex = zero
183 cex(1) = one
184 call dgesv(nmat1, 1, bloc, nmat1, ipiv, cex, nmat1, info)
185! call gjinv(nmat1, 1, bloc, cex, ok)
186 if (info.ne.0) then
187 nmat = 1
188 return
189 end if
190
191 xnew = zero
192 do i = 1, nmat
193 call daxpy(n,cex(i+1),x(:, i),1,xnew,1)
194 end do
195 nmat = nmat + 1
196
197 deallocate (bloc, cex, stat=istatus)
198end subroutine diis

201subroutine makeb(n, nmat, ndiis, e, b)
202 implicit none
203 integer, intent(in) :: n, nmat, ndiis
204 real(dp), dimension(n,ndiis), intent(in) :: e
205 real(dp), dimension(ndiis+1,ndiis+1), intent(inout) :: b
206
207 integer :: i
208 real(dp) :: bij
209 real(dp), parameter :: zero = 0.0d0, one = 1.0d0
210
211 ! 1st built
212 if (nmat.eq.1) then
213
214 ! [ 0 | 1 ]
215 ! b = [ --+---- ]
216 ! [ 1 | e*e ]
217
218 b(1,1) = zero
219 b(1,2) = one
220 b(2,1) = one
221 b(2,2) = dot_product(e(:,1),e(:,1))
222
223 ! subsequent builts
224 else
225
226 ! first, update the lagrangian line:
227 b(nmat+1,1) = one
228 b(1,nmat+1) = one
229
230 ! now, compute the new matrix elements:
231 !$omp parallel do default(none) shared(nmat,e,b) &
232 !$omp private(i,bij) schedule(dynamic)
233 do i = 1, nmat - 1
234 bij = dot_product(e(:,i), e(:,nmat))
235 b(nmat+1, i+1) = bij
236 b(i+1, nmat+1) = bij
237 end do
238 b(nmat+1,nmat+1) = dot_product(e(:,nmat),e(:,nmat))
239 end if
240end subroutine makeb

251subroutine jacobi_diis_external(params, constants, workspace, n, tol, rhs, x, n_iter, &
252 & x_rel_diff, matvec, dm1vec, norm_func)
253 type(ddx_params_type), intent(in) :: params
254 type(ddx_constants_type), intent(inout) :: constants
255 type(ddx_workspace_type), intent(inout) :: workspace
256 ! Inputs
257 integer, intent(in) :: n
258 real(dp), intent(in) :: tol
259 real(dp), dimension(n), intent(in) :: rhs
260 ! Outputs
261 real(dp), dimension(n), intent(inout) :: x
262 integer, intent(inout) :: n_iter
263 real(dp), intent(out) :: x_rel_diff(n_iter)
264 external :: matvec, dm1vec
265 procedure(norm_interface) :: norm_func
266 ! Local variables
267 integer :: it, nmat, istatus, lenb, nsph_u
268 real(dp) :: diff, norm, rel_diff = zero
269 logical :: dodiis
270 real(dp), allocatable :: x_new(:), y(:), x_diis(:,:), e_diis(:,:), bmat(:,:)
271 ! DIIS extrapolation flag
272 dodiis = (params%jacobi_ndiis.gt.0)
273 ! extrapolation required
274 if (dodiis) then
275 ! allocate workspaces
276 lenb = params % jacobi_ndiis + 1
277 allocate( x_diis(n,params % jacobi_ndiis), e_diis(n,params % jacobi_ndiis), bmat(lenb,lenb) , stat=istatus )
278 if (istatus .ne. 0) then
279 workspace % error_flag = 1
280 workspace % error_message = ' jacobi_diis: [1] failed allocation (diis)'
281 return
282 endif
283 ! initialize the number of points for diis to one.
284 ! note that nmat is updated by diis.
285 nmat = 1
286 endif
287 ! allocate workspaces
288 allocate( x_new(n), y(n) , stat=istatus )
289 if (istatus .ne. 0) then
290 workspace % error_flag = 1
291 workspace % error_message = ' jacobi_diis: [2] failed allocation (diis)'
292 return
293 endif
294 ! Jacobi iterations
295 do it = 1, n_iter
296 ! y = rhs - O x
297 call matvec(params, constants, workspace, x, y )
298 y = rhs - y
299 ! x_new = D^-1 y
300 call dm1vec(params, constants, workspace, y, x_new)
301 ! DIIS extrapolation
302 if (dodiis) then
303 x_diis(:,nmat) = x_new
304 e_diis(:,nmat) = x_new - x
305 call diis(n,nmat,params % jacobi_ndiis,x_diis,e_diis,bmat,x_new)
306 endif
307 !increment
308 x = x_new - x
309 ! warning: in the following, we use a quick and dirty hack to pass
310 ! to the norm function arrays double the size than in non ddlpb calculations
311 ! by defining a fake number of spheres with is n/nbasis (i.e., 2*nsph in
312 ! ddlpb).
313 nsph_u = n/constants % nbasis
314 diff = norm_func(params % lmax, constants % nbasis, nsph_u, x)
315 norm = norm_func(params % lmax, constants % nbasis, nsph_u, x_new)
316 ! compute rel_diff using the rule 0/0 = 0
317 if (diff.eq.zero .and. norm.eq.zero) then
318 rel_diff = zero
319 else
320 rel_diff = diff / norm
321 end if
322 x_rel_diff(it) = rel_diff
323 constants % inner_tol = max(rel_diff*sqrt(tol), tol/100.0d0)
324 ! update
325 x = x_new
326 ! EXIT Jacobi loop here
327 if (rel_diff .le. tol) then
328 n_iter = it
329 return
330 end if
331 enddo
332
333 workspace % error_flag = 1
334 workspace % error_message = ' jacobi_diis: [2] failed allocation (diis)'
335
336endsubroutine jacobi_diis_external
